017package com.github.siroshun09.configapi.api;
018
019import com.github.siroshun09.configapi.api.serializer.Serializer;
020import com.github.siroshun09.configapi.api.value.ConfigValue;
021import org.jetbrains.annotations.NotNull;
022import org.jetbrains.annotations.Nullable;
023import org.jetbrains.annotations.Unmodifiable;
024
025import java.util.Collections;
026import java.util.List;
027import java.util.Set;
028
029/**
030 * An interface that can get a value from a path.
031 */
032public interface Configuration {
033
034    /**
035     * The separator to split paths.
036     */
037    char PATH_SEPARATOR = '.';
038
039    /**
040     * The string representation of {@link Configuration#PATH_SEPARATOR}.
041     */
042    String PATH_SEPARATOR_STRING = String.valueOf(PATH_SEPARATOR);
043
044    /**
045     * Gets the object of the specified path.
046     * <p>
047     * If the object could not be obtained, this method returns {@code null}.
048     *
049     * @param path the path to get the object
050     * @return the object or {@code null}
051     */
052    @Nullable Object get(@NotNull String path);
053
054    /**
055     * Gets the object of the specified path.
056     * <p>
057     * If the object could not be obtained or deserialized,
058     * this method returns {@code null}.
059     *
060     * @param path       the path to get the object
061     * @param serializer the serializer to deserialize object
062     * @param <T>        the type of value
063     * @return the deserialized object or {@code null}
064     */
065    <T> @Nullable T get(@NotNull String path, @NotNull Serializer<T, ?> serializer);
066
067    /**
068     * Gets the object of the specified path.
069     *
070     * @param path the path to get the object
071     * @param def  the default object to return if the object could not be obtained
072     * @return the object
073     */
074    @NotNull Object get(@NotNull String path, @NotNull Object def);
075
076    /**
077     * Gets the object of the specified path.
078     * <p>
079     * The object to be returned will be deserialized using {@link Serializer}.
080     *
081     * @param path       the path to get the object
082     * @param serializer the serializer to deserialize the object
083     * @param def        the default object to return if the object could not be obtained or deserialized
084     * @param <T>        the type of value
085     * @return the deserialized object or default object
086     */
087    <T> @NotNull T get(@NotNull String path, @NotNull Serializer<T, ?> serializer, @NotNull T def);
088
089    /**
090     * Gets the value using {@link ConfigValue}.
091     *
092     * @param configValue the {@link ConfigValue} to get
093     * @param <T>         the type of value
094     * @return the object
095     */
096    <T> @NotNull T get(@NotNull ConfigValue<T> configValue);
097
098    /**
099     * Sets the object to the specified path.
100     * <p>
101     * If given object is {@code null},
102     * the current object of the path will be removed.
103     *
104     * @param path  the path to set or remove
105     * @param value the object to set or {@code null}
106     */
107    void set(@NotNull String path, @Nullable Object value);
108
109    /**
110     * Sets the object to the specified path.
111     * <p>
112     * The object to set will be serialized using {@link Serializer}
113     *
114     * @param path       the path to set
115     * @param value      the value to set
116     * @param serializer the serializer to serialize value
117     * @param <T>        the type of value
118     */
119    <T> void set(@NotNull String path, @NotNull T value, @NotNull Serializer<T, ?> serializer);
120
121    /**
122     * Gets the set of root paths included in this {@link Configuration}.
123     *
124     * @return set of root paths
125     * @deprecated use {@link Configuration#getKeyList()}
126     */
127    @Deprecated(since = "4.3.0", forRemoval = true)
128    @NotNull @Unmodifiable Set<String> getPaths();
129
130    /**
131     * Gets the list of root keys included in this {@link Configuration}.
132     *
133     * This method may not return to the correct order depending on the implementation.
134     *
135     * @return list of root keys
136     */
137    @NotNull @Unmodifiable List<String> getKeyList();
138
139    /**
140     * Gets the set of root objects included in this {@link Configuration}.
141     *
142     * @return set of root keys
143     */
144    @NotNull @Unmodifiable Set<Object> getValues();
145
146    /**
147     * Gets the {@link Configuration} section of the specified path.
148     * <p>
149     * If the {@link Configuration} section could not be obtained,
150     * this method returns {@code null}.
151     *
152     * @param path the path to get the {@link Configuration} section
153     * @return the {@link Configuration} section or {@code null}
154     */
155    @Nullable Configuration getSection(@NotNull String path);
156
157    /**
158     * Gets the list of the specified path.
159     * <p>
160     * If the list could not be obtained,
161     * this method returns {@link Collections#emptyList()}.
162     *
163     * @param path the path to get the list
164     * @return the list
165     */
166    @NotNull @Unmodifiable List<?> getList(@NotNull String path);
167
168    /**
169     * Gets the list of the specified path.
170     *
171     * @param path the path to get the list
172     * @param def  the default list to return if the list could not be obtained
173     * @return the list or default one
174     */
175    @NotNull @Unmodifiable List<?> getList(@NotNull String path, @NotNull List<?> def);
176
177    /**
178     * Gets the list of the specified path.
179     * <p>
180     * Each element in the list is converted by the {@link Serializer},
181     * and those that cannot be converted are removed from the returning list.
182     * <p>
183     * The list stored in this configuration is not changed.
184     *
185     * @param path       the path to get the list
186     * @param serializer the serializer to serialize elements
187     * @param <T>        the type of value
188     * @return the serialized list
189     */
190    @NotNull @Unmodifiable <T> List<T> getList(@NotNull String path, @NotNull Serializer<T, ?> serializer);
191
192    /**
193     * Gets the list of the specified path.
194     * <p>
195     * Each element in the list is converted by the {@link Serializer},
196     * and those that cannot be converted are removed from the list to be returned.
197     * <p>
198     * The list stored in this configuration will not be changed.
199     *
200     * @param path       the path to get the list
201     * @param serializer the serializer to serialize elements
202     * @param def        the default list to return if the list could not be obtained
203     * @param <T>        the type of value
204     * @return the serialized list or default one
205     */
206    @NotNull @Unmodifiable <T> List<T> getList(@NotNull String path,
207                                               @NotNull Serializer<T, ?> serializer, List<T> def);
208
209    /**
210     * Sets the list to specified path.
211     * <p>
212     * Each element in the list is converted by the {@link Serializer},
213     * and those that cannot be converted are removed from the list to be set.
214     * <p>
215     * The list given as an argument will not be changed.
216     *
217     * @param path       the path to set the list
218     * @param list       the list to set
219     * @param serializer the serializer to serialize elements
220     * @param <T>        the type of value
221     */
222    <T> void setList(@NotNull String path, @NotNull List<T> list, @NotNull Serializer<T, ?> serializer);
223
224    /**
225     * Gets the boolean of the specified path.
226     * <p>
227     * If the boolean could not be obtained, this method returns {@code false}.
228     *
229     * @param path the path to get the boolean
230     * @return the boolean
231     */
232    boolean getBoolean(@NotNull String path);
233
234    /**
235     * Gets the boolean of the specified path.
236     *
237     * @param path the path to get the boolean
238     * @param def  the default boolean to return if the boolean could not be obtained
239     * @return the boolean
240     */
241    boolean getBoolean(@NotNull String path, boolean def);
242
243    /**
244     * Gets the byte of the specified path.
245     * <p>
246     * If the byte could not be obtained, this method returns 0.
247     *
248     * @param path the path to get the byte
249     * @return the byte
250     */
251    byte getByte(@NotNull String path);
252
253    /**
254     * Gets the byte of the specified path.
255     *
256     * @param path the path to get the byte
257     * @param def  the default byte to return if the byte could not be obtained
258     * @return the byte
259     */
260    byte getByte(@NotNull String path, byte def);
261
262    /**
263     * Gets the double of the specified path.
264     * <p>
265     * If the double could not be obtained, this method returns 0.
266     *
267     * @param path the path to get the double
268     * @return the double
269     */
270    double getDouble(@NotNull String path);
271
272    /**
273     * Gets the double of the specified path.
274     *
275     * @param path the path to get the double
276     * @param def  the default double to return if the double could not be obtained
277     * @return the double
278     */
279    double getDouble(@NotNull String path, double def);
280
281    /**
282     * Gets the float of the specified path.
283     * <p>
284     * If the float could not be obtained, this method returns 0.
285     *
286     * @param path the path to get the float
287     * @return the float
288     */
289    float getFloat(@NotNull String path);
290
291    /**
292     * Gets the float of the specified path.
293     *
294     * @param path the path to get the float
295     * @param def  the default float to return if the float could not be obtained
296     * @return the float
297     */
298    float getFloat(@NotNull String path, float def);
299
300    /**
301     * Gets the integer of the specified path.
302     * <p>
303     * If the integer could not be obtained, this method returns 0.
304     *
305     * @param path the path to get the integer
306     * @return the integer
307     */
308    int getInteger(@NotNull String path);
309
310    /**
311     * Gets the integer of the specified path.
312     *
313     * @param path the path to get the integer
314     * @param def  the default integer to return if the integer could not be obtained
315     * @return the integer
316     */
317    int getInteger(@NotNull String path, int def);
318
319    /**
320     * Gets the long of the specified path.
321     * <p>
322     * If the long could not be obtained, this method returns 0.
323     *
324     * @param path the path to get the long
325     * @return the long
326     */
327    long getLong(@NotNull String path);
328
329    /**
330     * Gets the long of the specified path.
331     *
332     * @param path the path to get the long
333     * @param def  the default long to return if the long could not be obtained
334     * @return the long
335     */
336    long getLong(@NotNull String path, long def);
337
338    /**
339     * Gets the short of the specified path.
340     * <p>
341     * If the short could not be obtained, this method returns 0.
342     *
343     * @param path the path to get the short
344     * @return the short
345     */
346    short getShort(@NotNull String path);
347
348    /**
349     * Gets the short of the specified path.
350     *
351     * @param path the path to get the short
352     * @param def  the default short to return if the short could not be obtained
353     * @return the short
354     */
355    short getShort(@NotNull String path, short def);
356
357    /**
358     * Gets the string of the specified path.
359     * <p>
360     * If the string could not be obtained, this method returns 0.
361     *
362     * @param path the path to get the string
363     * @return the string
364     */
365    @NotNull String getString(@NotNull String path);
366
367    /**
368     * Gets the string of the specified path.
369     *
370     * @param path the path to get the string
371     * @param def  the default string to return if the string could not be obtained
372     * @return the string
373     */
374    @NotNull String getString(@NotNull String path, @NotNull String def);
375
376    /**
377     * Gets the boolean list of the specified path.
378     * <p>
379     * If the boolean list could not be obtained,
380     * this method returns {@link Collections#emptyList()}.
381     *
382     * @param path the path to get the boolean list
383     * @return the boolean list
384     */
385    @NotNull @Unmodifiable List<Boolean> getBooleanList(@NotNull String path);
386
387    /**
388     * Gets the boolean list of the specified path.
389     *
390     * @param path the path to get the boolean list
391     * @param def  the default boolean list to return if the boolean list could not be obtained
392     * @return the boolean list
393     */
394    @NotNull @Unmodifiable List<Boolean> getBooleanList(@NotNull String path, @NotNull List<Boolean> def);
395
396    /**
397     * Gets the byte list of the specified path.
398     * <p>
399     * If the byte list could not be obtained,
400     * this method returns {@link Collections#emptyList()}.
401     *
402     * @param path the path to get the byte list
403     * @return the byte list
404     */
405    @NotNull @Unmodifiable List<Byte> getByteList(@NotNull String path);
406
407    /**
408     * Gets the byte list of the specified path.
409     *
410     * @param path the path to get the byte list
411     * @param def  the default byte list to return if the byte list could not be obtained
412     * @return the byte list
413     */
414    @NotNull @Unmodifiable List<Byte> getByteList(@NotNull String path, @NotNull List<Byte> def);
415
416    /**
417     * Gets the double list of the specified path.
418     * <p>
419     * If the double list could not be obtained,
420     * this method returns {@link Collections#emptyList()}.
421     *
422     * @param path the path to get the double list
423     * @return the double list
424     */
425    @NotNull @Unmodifiable List<Double> getDoubleList(@NotNull String path);
426
427    /**
428     * Gets the double list of the specified path.
429     *
430     * @param path the path to get the double list
431     * @param def  the default double list to return if the double list could not be obtained
432     * @return the double list
433     */
434    @NotNull @Unmodifiable List<Double> getDoubleList(@NotNull String path, @NotNull List<Double> def);
435
436    /**
437     * Gets the float list of the specified path.
438     * <p>
439     * If the float list could not be obtained,
440     * this method returns {@link Collections#emptyList()}.
441     *
442     * @param path the path to get the float list
443     * @return the float list
444     */
445    @NotNull @Unmodifiable List<Float> getFloatList(@NotNull String path);
446
447    /**
448     * Gets the float list of the specified path.
449     *
450     * @param path the path to get the float list
451     * @param def  the default float list to return if the float list could not be obtained
452     * @return the float list
453     */
454    @NotNull @Unmodifiable List<Float> getFloatList(@NotNull String path, @NotNull List<Float> def);
455
456    /**
457     * Gets the integer list of the specified path.
458     * <p>
459     * If the integer list could not be obtained,
460     * this method returns {@link Collections#emptyList()}.
461     *
462     * @param path the path to get the integer list
463     * @return the integer list
464     */
465    @NotNull @Unmodifiable List<Integer> getIntegerList(@NotNull String path);
466
467    /**
468     * Gets the integer list of the specified path.
469     *
470     * @param path the path to get the integer list
471     * @param def  the default integer list to return if the integer list could not be obtained
472     * @return the integer list
473     */
474    @NotNull @Unmodifiable List<Integer> getIntegerList(@NotNull String path, @NotNull List<Integer> def);
475
476    /**
477     * Gets the long list of the specified path.
478     * <p>
479     * If the long list could not be obtained,
480     * this method returns {@link Collections#emptyList()}.
481     *
482     * @param path the path to get the long list
483     * @return the long list
484     */
485    @NotNull @Unmodifiable List<Long> getLongList(@NotNull String path);
486
487    /**
488     * Gets the long list of the specified path.
489     *
490     * @param path the path to get the long list
491     * @param def  the default long list to return if the long list could not be obtained
492     * @return the long list
493     */
494    @NotNull @Unmodifiable List<Long> getLongList(@NotNull String path, @NotNull List<Long> def);
495
496    /**
497     * Gets the short list of the specified path.
498     * <p>
499     * If the short list could not be obtained,
500     * this method returns {@link Collections#emptyList()}.
501     *
502     * @param path the path to get the short list
503     * @return the short list
504     */
505    @NotNull @Unmodifiable List<Short> getShortList(@NotNull String path);
506
507    /**
508     * Gets the short list of the specified path.
509     *
510     * @param path the path to get the short list
511     * @param def  the default short list to return if the short list could not be obtained
512     * @return the short list
513     */
514    @NotNull @Unmodifiable List<Short> getShortList(@NotNull String path, @NotNull List<Short> def);
515
516    /**
517     * Gets the string list of the specified path.
518     * <p>
519     * If the string list could not be obtained,
520     * this method returns {@link Collections#emptyList()}.
521     *
522     * @param path the path to get the string list
523     * @return the string list
524     */
525    @NotNull @Unmodifiable List<String> getStringList(@NotNull String path);
526
527    /**
528     * Gets the string list of the specified path.
529     *
530     * @param path the path to get the string list
531     * @param def  the default string list to return if the string list could not be obtained
532     * @return the string list
533     */
534    @NotNull @Unmodifiable List<String> getStringList(@NotNull String path, @NotNull List<String> def);
535
536    /**
537     * Gets the byte array.
538     * <p>
539     * This method decodes a base64 string into a byte array using {@link java.util.Base64#getDecoder()}.
540     *
541     * @param path the path to get the byte array
542     * @return the decoded byte array or the empty byte array if could not decode base64 string
543     */
544    byte[] getBytes(@NotNull String path);
545
546    /**
547     * Sets the byte array.
548     * <p>
549     * This method encodes a byte array into a base64 string using {@link java.util.Base64#getEncoder}.
550     *
551     * @param path  the path to set the byte array
552     * @param bytes the byte array to set
553     */
554    void setBytes(@NotNull String path, byte[] bytes);
